# 项目目录结构


├── docs                                    // makdown文档编辑
   └── .vuepress                            // button组件
     ├── config.js                          // 文档控制器
     ├── override.styl                      // 文档样式替换
   │── guide                                // 新手指南
   ├── README.md                            // 文档首页

├── public                                  // 公共
├── src                                     // 项目源码
    └── api                                 // api统一管理
    └── assets                              // 样式管理
        ├── svg-icons                       // svg图片
    └── components                          // 存放无状态组件
    └── layout                              // layout布局
    └── menu                                // 菜单布局
    └── pages                               // 页面源码
    └── plugin                              // 插件管理
    └── router                              // 路由控制中心
    └── store                               // vuex控制中心
    └── app.vue                             // vue页面挂载 全局页面
    └── main.js                             // js 入口文件
    └── setting.js                          // 初始化设置文件
├── tests                                   // 单元测试
├── .eslintrc.js                            // eslint 规则设置
├── package.json                            // 插件表
├── vue.config.js                           // webpack 基础设置文件
